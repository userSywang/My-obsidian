<%*
// 📓 追加股票日记（使用 Obsidian API 直接操作文件）
const date = await tp.system.prompt("📅 日期", tp.date.now("YYYY-MM-DD"));
const content = await tp.system.prompt("📝 今日记录");

// 找到日记文件
const diaryFile = tp.file.find_tfile("股票学习/随记/股票日记.md");

if (diaryFile) {
    // 读取当前内容
    const currentContent = await app.vault.read(diaryFile);
    // 在末尾追加
    const newEntry = "\n### " + date + "\n\n" + content + "\n";
    await app.vault.modify(diaryFile, currentContent + newEntry);
    new Notice("✅ 日记已追加到 股票日记.md");
} else {
    new Notice("❌ 找不到日记文件");
}
_%>
