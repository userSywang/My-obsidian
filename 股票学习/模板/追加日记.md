<%*
try {
    // 📓 追加股票日记（最新在最上面）
    const date = await tp.system.prompt("📅 日期", tp.date.now("YYYY-MM-DD"));
    if (!date) {
        new Notice("❌ 已取消");
        return;
    }
    
    const content = await tp.system.prompt("📝 今日记录");
    if (!content || content.trim() === "") {
        new Notice("❌ 内容不能为空");
        return;
    }
    
    // 查找日记文件（使用 Obsidian API）
    const diaryFile = app.vault.getAbstractFileByPath("随记/股票日记.md");
    
    if (diaryFile) {
        let currentContent = await app.vault.read(diaryFile);
        
        // 新日记条目
        const newEntry = "\n### " + date + "\n\n" + content + "\n";
        
        // 在第一个 "### 2026/" 前面插入（保持最新在最上）
        const insertMarker = "\n### 2026/";
        const insertPos = currentContent.indexOf(insertMarker);
        
        if (insertPos !== -1) {
            currentContent = currentContent.slice(0, insertPos) + newEntry + currentContent.slice(insertPos);
        } else {
            // 容错：追加到末尾
            currentContent = currentContent + newEntry;
        }
        
        await app.vault.modify(diaryFile, currentContent);
        new Notice("✅ 日记已追加（" + date + "）");
    } else {
        new Notice("❌ 找不到随记/股票日记.md");
    }
} catch (e) {
    new Notice("❌ 错误: " + e.message);
}
_%>
